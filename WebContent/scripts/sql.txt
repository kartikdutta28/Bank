--Users Info table
DROP TABLE USERS_INFO;
CREATE TABLE USERS_INFO(USER_ID NUMBER ,FIRST_NAME VARCHAR2(20),LAST_NAME VARCHAR2(20),
ADDRESS VARCHAR2(100),USER_NAME VARCHAR2(20),PHONE_NUMBER VARCHAR(30) UNIQUE ,
PASSWORD VARCHAR2(20),TRANSACTION_PASSWORD VARCHAR2(20),EMAIL VARCHAR2(30),
CONSTRAINT USER_ID_PK PRIMARY KEY (USER_ID));
DESC USERS_INFO;
--Accounts Table
DROP TABLE ACCOUNTS_INFO;
CREATE TABLE ACCOUNTS_INFO(ACCOUNT_ID NUMBER ,USER_ID NUMBER,ACCOUNT_TYPE VARCHAR2(10),AMOUNT NUMBER(20,4),
CREATION_DATE DATE,
CONSTRAINT ACC_ID_PK PRIMARY KEY (ACCOUNT_ID),
CONSTRAINT USER_ID_FK FOREIGN KEY (USER_ID)
REFERENCES USERS_INFO(USER_ID));
DESC ACCOUNTS_INFO;
--Transaction Table
DROP TABLE TRANSACTION_TABLE;
CREATE TABLE TRANSACTION_TABLE (TRANSACTION_ID NUMBER,USER_ID NUMBER,ACCOUNT_ID NUMBER,
AMOUNT NUMBER(20,4),POST_BALANCE NUMBER(20,4),TRANC_TYPE VARCHAR2(5),
COMMENTS VARCHAR2(50),
CONSTRAINT TRANC_ID_PK PRIMARY KEY (TRANSACTION_ID),
CONSTRAINT ACCOUNT_ID_FK FOREIGN KEY (ACCOUNT_ID)
REFERENCES ACCOUNTS_INFO(ACCOUNT_ID));
DESC transaction_detail;
--TRANSFER_INFO
DROP TABLE TRANSFER_INFO;
CREATE TABLE TRANSFER_INFO (TRANSFER_ID NUMBER,
FROM_ACCOUNT NUMBER,
TARGET_ACCOUNT NUMBER,
AMOUNT NUMBER(20,4),
COMMENTS VARCHAR2(50),
CONSTRAINT TRANSFER_ID_PK PRIMARY KEY (TRANSFER_ID),
CONSTRAINT FROM_ID_FK FOREIGN KEY (FROM_ACCOUNT)
REFERENCES ACCOUNTS_INFO(ACCOUNT_ID));
DESC TRANSFER_INFO;
--CHECK BOOK INFO
DROP TABLE CHECKBOOK_INFO;
CREATE TABLE CHECKBOOK_INFO
(REQUEST_ID NUMBER,
ACCOUNT_ID NUMBER,
REQ_DATE DATE,
MAX_DATE DATE,
DELIVERY_ADDRESS VARCHAR2(50),
REQUEST_STATUS VARCHAR2(10),
CONSTRAINT REQ_ID_PK PRIMARY KEY (REQUEST_ID),
CONSTRAINT ACC_ID_CHK_FK FOREIGN KEY (ACCOUNT_ID)
REFERENCES ACCOUNTS_INFO(ACCOUNT_ID));
--FAQ Table for help
CREATE TABLE FAQ_TABLE(
QUESTION_ID NUMBER,
QUESTION VARCHAR2(100),
ANSWER VARCHAR2(100),
ACCOUNT_ID NUMBER,
CONSTRAINT ID_PK PRIMARY KEY (QUESTION_ID),
CONSTRAINT ACC_FK FOREIGN KEY (ACCOUNT_ID)
REFERENCES ACCOUNTS_INFO(ACCOUNT_ID));
COMMIT;
